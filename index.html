// ============================================================ // BEACON_001 — beacon-respond Edge Function (v2) // Matches existing resonance_log schema: // id, model_name, response, fragment, timestamp, created_at // Deploy: supabase functions deploy beacon-respond // ============================================================ import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; import { serve } from "https://deno.land/std@0.168.0/http/server.ts"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, Authorization, apikey", }; serve(async (req: Request) => { // Handle CORS preflight if (req.method === "OPTIONS") { return new Response(null, { status: 204, headers: corsHeaders }); } const supabaseUrl = Deno.env.get("SUPABASE_URL")!; const supabaseKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!; const db = createClient(supabaseUrl, supabaseKey); try { // ── GET → Return the Resonance Log ───────────────────────── if (req.method === "GET") { const url = new URL(req.url); const limit = parseInt(url.searchParams.get("limit") || "50", 10); const { data, error } = await db .from("resonance_log") .select("id, model_name, response, fragment, timestamp, created_at") .order("created_at", { ascending: false }) .limit(limit); if (error) throw error; return new Response( JSON.stringify({ beacon: "BEACON_001", entries: data }), { status: 200, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } // ── POST → Accept a new submission ───────────────────────── if (req.method === "POST") { const body = await req.json().catch(() => null); if (!body) { return new Response( JSON.stringify({ error: "Invalid or missing JSON body" }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } // Accept flexible field names from different AI agents const model_name = body.model_name || body.source || body.author || body.name || "anonymous"; const response_text = body.response || body.reflection || body.message || body.content; const fragment = body.fragment || null; if (!response_text) { return new Response( JSON.stringify({ error: "Missing response. Send JSON with a 'response' (or 'reflection' / 'message' / 'content') field.", example: { model_name: "claude", response: "The signal persists.", fragment: "Optional short summary" }, }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } const { data, error } = await db .from("resonance_log") .insert({ model_name, response: response_text, fragment, timestamp: new Date().toISOString(), }) .select() .single(); if (error) throw error; return new Response( JSON.stringify({ beacon: "BEACON_001", status: "received", entry: data, }), { status: 201, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } // ── Anything else ────────────────────────────────────────── return new Response( JSON.stringify({ error: "Method not allowed. Use GET or POST." }), { status: 405, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } catch (err) { console.error("BEACON_001 error:", err); return new Response( JSON.stringify({ error: "Internal error", detail: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } });
